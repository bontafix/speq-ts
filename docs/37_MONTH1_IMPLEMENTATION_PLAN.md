import { Redis } from 'ioredis';
import type { WizardSession } from './types';
import type { SessionStore } from './session.store.interface';

export class RedisSessionStore implements SessionStore {
  private redis: Redis;
  private ttl: number;

  constructor() {
    this.redis = new Redis({
      host: process.env.REDIS_HOST || 'localhost',
      port: Number(process.env.REDIS_PORT) || 6379,
      password: process.env.REDIS_PASSWORD || undefined,
      db: Number(process.env.REDIS_DB) || 0,
      retryStrategy: (times) => {
        const delay = Math.min(times * 50, 2000);
        return delay;
      },
    });

    this.ttl = Number(process.env.REDIS_TTL) || 86400; // 24 —á–∞—Å–∞

    // –û–±—Ä–∞–±–æ—Ç–∫–∞ –æ—à–∏–±–æ–∫
    this.redis.on('error', (err) => {
      console.error('[Redis] Connection error:', err);
    });

    this.redis.on('connect', () => {
      console.log('[Redis] Connected successfully');
    });
  }

  private getKey(telegramId: number): string {
    return `session:${telegramId}`;
  }

  async get(telegramId: number): Promise<WizardSession | null> {
    try {
      const data = await this.redis.get(this.getKey(telegramId));
      if (!data) return null;
      return JSON.parse(data) as WizardSession;
    } catch (error) {
      console.error(`[Redis] Failed to get session ${telegramId}:`, error);
      return null;
    }
  }

  async set(session: WizardSession): Promise<void> {
    try {
      const data = JSON.stringify(session);
      await this.redis.setex(
        this.getKey(session.telegramId),
        this.ttl,
        data
      );
    } catch (error) {
      console.error(`[Redis] Failed to set session ${session.telegramId}:`, error);
      throw error;
    }
  }

  async delete(telegramId: number): Promise<void> {
    try {
      await this.redis.del(this.getKey(telegramId));
    } catch (error) {
      console.error(`[Redis] Failed to delete session ${telegramId}:`, error);
    }
  }

  async exists(telegramId: number): Promise<boolean> {
    try {
      const result = await this.redis.exists(this.getKey(telegramId));
      return result === 1;
    } catch (error) {
      console.error(`[Redis] Failed to check session ${telegramId}:`, error);
      return false;
    }
  }

  async clearExpired(): Promise<number> {
    // Redis –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ —É–¥–∞–ª—è–µ—Ç –∏—Å—Ç–µ–∫—à–∏–µ –∫–ª—é—á–∏

  }

  async disconnect(): Promise<void> {

  }
}
